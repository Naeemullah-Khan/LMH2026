\section{Challenges \& Current Research}

\begin{frame}{}
    \LARGE \textbf{Challenges \& Current Research}
\end{frame}

\begin{frame}[allowframebreaks]{Open Research Challenges in Speech & Language Systems}

\begin{itemize}
    \item \textbf{Robustness:} Handling noise, accents, and low-quality microphones.  

    \item \textbf{Multilinguality \& Code-Switching:} Supporting diverse populations and hybrid language input.  

    \item \textbf{Low-Resource Speech:} Many languages lack sufficient training data for ASR/TTS models.  

    \item \textbf{Expressive Speech Synthesis:} Capturing prosody, emotion, and enabling style transfer.  

    \item \textbf{Evaluation:} Moving beyond BLEU/WER to semantic, human-centered, and LLM-based metrics.  

    \item \textbf{Ethics:} Addressing risks like deepfake voices and protecting privacy in speech datasets.
\end{itemize}

\end{frame}
