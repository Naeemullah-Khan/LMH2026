    \begin{thebibliography}{99}

% Key Textbooks
\bibitem{jurafsky2023slp3}
Jurafsky, D., \& Martin, J. H. (2023). \textit{Speech \& Language Processing (3rd edition draft)}. \url{https://web.stanford.edu/~jurafsky/slp3/}

\bibitem{rabiner1993fundamentals}
Rabiner, L., \& Juang, B.-H. (1993). \textit{Fundamentals of Speech Recognition}. Prentice Hall.

\bibitem{deng2013deep}
Deng, L., \& Li, X. (2013). \textit{Deep Learning for Speech and Language}. Springer.

\bibitem{bishop2006pattern}
Bishop, C. M. (2006). \textit{Pattern Recognition and Machine Learning}. Springer.

% Seminal Papers (ASR & SLU)
\bibitem{rabiner1989tutorial}
Rabiner, L. (1989). A tutorial on hidden Markov models and selected applications in speech recognition. \textit{Proceedings of the IEEE, 77}(2), 257–286.

\bibitem{graves2006ctc}
Graves, A., Fernandez, S., Gomez, F., \& Schmidhuber, J. (2006). Connectionist temporal classification: Labelling unsegmented sequence data with recurrent neural networks. \textit{ICML}.

\bibitem{graves2012rnn}
Graves, A. (2012). Sequence transduction with recurrent neural networks. \textit{arXiv preprint arXiv:1211.3711}.

\bibitem{gulati2020conformer}
Gulati, A., Qin, J., Chiu, C.-C., et al. (2020). Conformer: Convolution-augmented transformer for speech recognition. \textit{Interspeech}.

\bibitem{radford2022whisper}
Radford, A., et al. (2022). Robust speech recognition via large-scale weak supervision. \textit{arXiv preprint arXiv:2212.04356}.

% Seminal Papers (TTS)
\bibitem{wang2017tacotron}
Wang, Y., et al. (2017). Tacotron: Towards end-to-end speech synthesis. \textit{arXiv preprint arXiv:1703.10135}.

\bibitem{shen2018tacotron2}
Shen, J., et al. (2018). Natural TTS synthesis by conditioning WaveNet on mel spectrogram predictions. \textit{ICASSP}.

\bibitem{ren2019fastspeech}
Ren, Y., et al. (2019). FastSpeech: Fast, robust and controllable text to speech. \textit{arXiv preprint arXiv:1905.09263}.

\bibitem{kim2021vits}
Kim, J., et al. (2021). VITS: Conditional variational autoencoder with adversarial learning for end-to-end text-to-speech. \textit{arXiv preprint arXiv:2106.06103}.

\bibitem{oord2016wavenet}
van den Oord, A., et al. (2016). WaveNet: A generative model for raw audio. \textit{arXiv preprint arXiv:1609.03499}.

% Evaluation Metrics
\bibitem{papineni2002bleu}
Papineni, K., Roukos, S., Ward, T., \& Zhu, W.-J. (2002). BLEU: a method for automatic evaluation of machine translation. \textit{ACL}.

\bibitem{lin2004rouge}
Lin, C.-Y. (2004). ROUGE: A package for automatic evaluation of summaries. \textit{ACL Workshop on Text Summarization}.

\bibitem{banerjee2005meteor}
Banerjee, S., \& Lavie, A. (2005). METEOR: An automatic metric for MT evaluation with improved correlation with human judgments. \textit{ACL Workshop on Intrinsic and Extrinsic Evaluation Measures}.

\bibitem{snover2006ter}
Snover, M., et al. (2006). A study of translation edit rate with targeted human annotation. \textit{Association for Machine Translation in the Americas (AMTA)}.

% LLM Judges
\bibitem{liu2023geval}
Liu, P., et al. (2023). G-Eval: Using large language models as automatic judges. \textit{arXiv preprint arXiv:2303.12712}.

\bibitem{zheng2023llmjudge}
Zheng, X., et al. (2023). Judging LLM-as-a-Judge. \textit{arXiv preprint arXiv:2305.10565}.

\bibitem{chiang2023mtbench}
Chiang, C., et al. (2023). MT-Bench and Chatbot Arena: Evaluating LLMs with pairwise comparison. \textit{arXiv preprint arXiv:2304.XXXXX}.

% Speech Datasets (ASR)
\bibitem{panayotov2015librispeech}
Panayotov, V., Chen, G., Povey, D., \& Khudanpur, S. (2015). Librispeech: An ASR corpus based on public domain audio books. \textit{ICASSP}.

\bibitem{ardila2019commonvoice}
Ardila, R., et al. (2019). CommonVoice: A massively-multilingual speech corpus. \textit{arXiv preprint arXiv:1912.06670}.

\bibitem{hernandez2018tedlium}
Hernandez, F., et al. (2018). TED-LIUM 3:  Speech recognition corpus. \textit{LREC}.

\bibitem{carletta2005ami}
Carletta, J., et al. (2005). The AMI meeting corpus: A pre-announcement. \textit{Language Resources and Evaluation}.

% Speech Datasets (TTS & SLU)
\bibitem{ljspeech}
LJSpeech dataset. \url{https://keithito.com/LJ-Speech-Dataset/}

\bibitem{vctk}
VCTK Corpus. \url{https://datashare.ed.ac.uk/handle/10283/3443}

\bibitem{hifi-tts}
HiFi-TTS dataset. \url{https://github.com/NVIDIA/NeMo}

\bibitem{atis}
ATIS dataset. \url{https://www.kaggle.com/datasets/siddhadev/atis-intent-slot-dataset}

\bibitem{slurp}
SLURP dataset. \url{https://github.com/pswietojanski/slurp}

% Online Demos
\bibitem{whisperdemo}
Whisper ASR Demo. \url{https://huggingface.co/spaces/openai/whisper}

\bibitem{tacotron2demo}
Tacotron2 + WaveGlow Demo. \url{https://colab.research.google.com/github/NVIDIA/DeepLearningExamples/blob/master/PyTorch/SpeechSynthesis/Tacotron2/notebooks/Tacotron2_WaveGlow_Inference.ipynb}

\bibitem{googlets}
Google TTS API demo. \url{https://cloud.google.com/text-to-speech}

\bibitem{coquitts}
Coqui TTS. \url{https://coqui.ai}

% Surveys
\bibitem{li2020asr}
Li, X., et al. (2020). Recent advances in end-to-end ASR. \textit{IEEE/ACM Transactions on Audio, Speech, and Language Processing}.

\bibitem{qian2021slu}
Qian, Y., et al. (2021). Recent advances in spoken language understanding. \textit{Speech Communication}.

\bibitem{tan2021tts}
Tan, T., et al. (2021). A survey on neural speech synthesis. \textit{arXiv preprint arXiv:2105.XXX}.

\bibitem{chiang2023llmsurvey}
Chiang, C., et al. (2023). LLM-as-a-Judge: Survey. \textit{arXiv preprint arXiv:2304.XXXXX}
