\section{Impact of Attention}
\begin{frame}{Impact of Attention}
  \begin{itemize}
    \item Improves translation accuracy
    \item Helps handle \textbf{longer sequences}
    \item Interpretable: shows what the model is focusing on
    \item Forms the foundation of \textbf{Transformer models}
  \end{itemize}
  \vspace{1em}
  \textbf{Led to development of:}
  \begin{itemize}
    \item[$\blacktriangleright$] Bahdanau Attention (Additive, 2014)
    \item[$\blacktriangleright$] Luong Attention (Multiplicative, 2015)
    \item[$\blacktriangleright$] Self-Attention and Transformers (2017+)
  \end{itemize}
\end{frame}
