\begin{frame}{Limitations of LLMs}
  \begin{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] High inference costs \& latency
    \item[\textcolor{red}{$\blacktriangleright$}] Bias and toxic outputs
    \item[\textcolor{red}{$\blacktriangleright$}] Hallucination and factual inconsistency
    \item[\textcolor{red}{$\blacktriangleright$}] Limited interpretability
    \item[\textcolor{red}{$\blacktriangleright$}] Require vast pre-training data
    \item[\textcolor{red}{$\blacktriangleright$}] Poor domain generalization (medical, legal, etc.)
  \end{itemize}
\end{frame}
