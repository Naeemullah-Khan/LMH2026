      \end{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Memory-Augmented Models:}
      \begin{itemize}
        \item Retrieval-augmented generation (RAG)
        \item Memory layers (e.g., RetNet)
      \end{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Chunking + Re-ranking:}
      \begin{itemize}

      \end{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Recurrence or State Passing:}
      \begin{itemize}
        \item Transformer-XL, RWKV (RNN-inspired)
      \end{itemize}
  \end{itemize}
\end{frame}
