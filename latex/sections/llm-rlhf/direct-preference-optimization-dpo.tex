\begin{frame}[allowframebreaks]{Direct Preference Optimization (DPO)}
  \begin{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] \textbf{New method:} Bypasses RL and reward model training entirely.
    \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Key Idea:} Optimize the policy directly from human preference data.
    \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Reference:} Rafailov et al., 2023.
  \end{itemize}
  
  \vspace{0.6em}
  \textbf{DPO Loss:}
  
  \[
  L_{\mathrm{DPO}} = \log \frac{
      \exp\left(\beta \log \frac{\pi(y^+)}{\pi_0(y^+)}\right)
    }{
      \exp\left(\beta \log \frac{\pi(y^+)}{\pi_0(y^+)}\right) + \exp\left(\beta \log \frac{\pi(y^-)}{\pi_0(y^-)}\right)
    }
  \]
  
  \footnotesize
  where $y^+$ is the preferred response, $y^-$ is the dispreferred response, $\pi$ is the current policy, $\pi_0$ is the reference (SFT) policy, and $\beta$ is a temperature parameter.

  \framebreak
  \normalsize
  \textcolor{red}{\textbf{Benefits:}}
  
  \begin{itemize}
    \item Simpler and more stable than PPO
    \item No need for reward model training or RL rollouts
  \end{itemize}
\end{frame}
