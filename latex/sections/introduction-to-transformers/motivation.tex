\begin{frame}{Motivation}
  \begin{itemize}
    \item[$\blacktriangleright$] \textbf{Traditional RNNs/LSTMs suffer from:}
    \begin{itemize}
      \item Sequential computation $\rightarrow$ slow training
      \item Vanishing gradients $\rightarrow$ poor long-range dependencies
    \end{itemize}
    \vspace{0.8em}
    \item[$\blacktriangleright$] \textbf{Need for a model that:}
    \begin{itemize}
      \item Handles global context
      \item Is parallelizable
      \item Supports scalability
    \end{itemize}
  \end{itemize}
\end{frame}
