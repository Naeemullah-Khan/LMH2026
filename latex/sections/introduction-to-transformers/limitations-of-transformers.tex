\begin{frame}{Limitations of Transformers}
  \begin{itemize}
    \item[$\blacktriangleright$] High compute and memory requirements
    \item[$\blacktriangleright$] Poor extrapolation capabilities
    \item[$\blacktriangleright$] Limitations due to hard-coded positional encoding
    \item[$\blacktriangleright$] Quadratic attention cost ($O(n^2)$)
    \item[$\blacktriangleright$] Lack of reasoning and grounding
  \end{itemize}
\end{frame}
