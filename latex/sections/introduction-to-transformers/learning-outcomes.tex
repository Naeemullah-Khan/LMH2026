\begin{frame}{Learning Outcomes}
  \textbf{After this session, you should be able to:}
  \begin{itemize}
    \item[$\blacktriangleright$] Understand the self-attention mechanism
    \item[$\blacktriangleright$] Learn how multi-head attention works and why it helps
    \item[$\blacktriangleright$] Grasp the role of positional encoding
    \item[$\blacktriangleright$] Deconstruct transformer blocks and variants
    \item[$\blacktriangleright$] Compare Pre-LayerNorm vs.\ Post-LayerNorm
    \item[$\blacktriangleright$] Evaluate LayerNorm vs.\ RMSNorm
    \item[$\blacktriangleright$] Explore causal masking in decoder-only transformers
    \item[$\blacktriangleright$] Understand tradeoffs and future directions
  \end{itemize}
\end{frame}
