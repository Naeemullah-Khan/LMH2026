\begin{frame}[t]{Why Prompting and RAG?}
  \begin{itemize}
    \item LLMs (Large Language Models) are powerful but have limitations:
    \begin{itemize}
      \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Static knowledge} (limited to training data)
      \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Prone to hallucinations}
      \item[\textcolor{red}{$\blacktriangleright$}] \textbf{Require careful input phrasing}
    \end{itemize}
    \vspace{1em}
    \item Prompting helps guide LLM behavior.
    \vspace{1em}
    \item RAG (Retrieval-Augmented Generation) addresses knowledge limitations by combining LLMs with external retrieval.
  \end{itemize}
\end{frame}

\begin{frame}{Real-World Motivation}
  Chatbots, search engines, and coding assistants require:
  \begin{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] Accurate, real-time information
    \item[\textcolor{red}{$\blacktriangleright$}] Robust language understanding
  \end{itemize}

  \vspace{1em}
  Prompting combined with RAG enables systems that are:
  \begin{itemize}
    \item[\textcolor{red}{$\blacktriangleright$}] Customizable
    \item[\textcolor{red}{$\blacktriangleright$}] Scalable
    \item[\textcolor{red}{$\blacktriangleright$}] Easily updatable
  \end{itemize}
\end{frame}
